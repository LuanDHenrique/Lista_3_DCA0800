#include <stdio.h>

int main(void) {
  int i, x, cont1=0, cont2=0;

  for(i=1; i<=10; i++){
  printf("Digite um numero: ");
  scanf("%d", &x);

  if((x>=10)&&(x<=50)){
    cont1=cont1+1;
  }else{
    cont2=cont2+1;
   }
  }
  if(cont1==0){
  printf("Não existe nenhum numero dentro do intervalo entre 10 e 50.\n");
  printf("Existem %d numeros fora do intervalo entre 10 e 50.", cont2);
  }
  if((cont1>0)&&(cont1<10)){
  printf("Existem %d numeros dentro do intervalo entre 10 e 50.\n", cont1);
  printf("Existem %d numeros fora do intervalo entre 10 e 50.", cont2);
  }
  if(cont1==10){
  printf("Existem %d numeros dentro do intervalo entre 10 e 50.\n", cont1);
  printf("Não existe nenhum numero fora do intervalo entre 10 e 50.");
  }
  return 0;
}
